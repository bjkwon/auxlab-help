[FUNCTION NAME] isstring
[STATIC] 0
[FUNCTION DESCRIPTION] Checks if the input is string.
[PARAMETERS]  x
[PARAM TYPE DELIM BY TAB]  anything
[PARAM DESCRIPTION--MULTILINES]  
[PARAM UNIT--MULTILINES]  n/a
[NOTES]  
[EXAMPLES] x="bj kwon";
x.isstring &rarr; true 
[ALGORITHM]  
[SEE ALSO]  
[OUTPUT ARGS BY TAB] y
[OUTPUT ARGS--MULTILINES] #q1 is true if #p1 is string.
