[FUNCTION NAME] mod
[OPERATOR] %
[STATIC] 0
[FUNCTION DESCRIPTION]  Remainder after division (modulo operation)
[PARAMETERS]  x div
[PARAM TYPE DELIM BY TAB]    SCAL, VCT or AUDIO	SCAL
[PARAM DESCRIPTION--MULTILINES]  n/a
n/a
[PARAM UNIT--MULTILINES]  n/a
n/a
[NOTES]  
[EXAMPLES]  mod(17,5) %rarr; 2
mod(-1,3) %rarr; -1
(12).mod(5) %rarr; 2
[SEE ALSO]  
[ALGORITHM]  
[OUTPUT ARGS BY TAB] y
[OUTPUT ARGS--MULTILINES] #q1 is the remainder after division of #p1 by #p2
